"""Classes e metodos herdados."""


class Passaro:
    def __init__(self) -> None:
        pass

    def voar(self) -> None:
        print("Voando...")

    def emitir_som(self) -> None:
        pass


class Pato(Passaro):
    def __init__(self) -> None:
        print("Pato")

    def emitir_som(self):
        print("Pato emitindo som...")
        print("Quack Quack")


class Pardal(Passaro):
    def __init__(self) -> None:
        print("Pardal")

    def emitir_som(self):
        print("Pardal emitindo som...")
        print("Piu Piu")


# Inst√¢nciando as classes
pato = Pato()
pato.voar()
pato.emitir_som()

pardal = Pardal()
pardal.voar()
pardal.emitir_som()
